            document.getElementById('result-title').textContent = randomItem.title;
            
            const typeBadge = document.getElementById('result-type');
            typeBadge.className = 'px-3 py-1 rounded-full text-sm'; 

            let categoryName = '';
            if (category === 'movies') {
                typeBadge.textContent = 'فيلم';
                typeBadge.classList.add('bg-purple-500', 'text-white');
                categoryName = 'الأفلام';
            } else if (category === 'series') {
                typeBadge.textContent = 'مسلسل';
                typeBadge.classList.add('bg-blue-500', 'text-white');
                categoryName = 'المسلسلات';
            } else if (category === 'plays') {
                typeBadge.textContent = 'مسرحية';
                typeBadge.classList.add('bg-green-500', 'text-white');
                categoryName = 'المسرحيات';
            }
            
            const content = document.getElementById('result-content');
            let detailsHTML = `
                <p class="mb-2"><span class="font-semibold details-label">سنة الإنتاج:</span> ${randomItem.year || 'غير متوفر'}</p>
                <p class="mb-2"><span class="font-semibold details-label">أبرز الأبطال:</span> ${(randomItem.cast && randomItem.cast.length > 0) ? randomItem.cast.join('، ') : 'غير متوفر'}</p>
                <p class="mb-3"><span class="font-semibold details-label">نبذة:</span> ${randomItem.description || 'لا توجد نبذة متوفرة حاليًا.'}</p>
                <hr class="my-3 border-gray-700">
            `;

            content.innerHTML = `
                ${detailsHTML}
                <p class="mb-3">تهانينا! لقد تم اختيار <span class="font-bold">${randomItem.title}</span> لك كتحدي فني.</p>
                <p class="mb-3">هذا العمل يعتبر من ${categoryName} الصعبة التي قد تتطلب تركيزًا عاليًا لفهم جميع تفاصيلها.</p>
                <p>نتمنى لك مشاهدة ممتعة وتحديًا فنيًا رائعًا!</p>
            `;
